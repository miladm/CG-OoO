template<class T >
bool __x_parse_range ( const std::string &  str,

) 
{
    std::vector<std::string> values;
    tokenize(values, str, ':');

    T thefirst, thelast;

    if (values.size() != 2)
        return false;

    std::string s = values[0];
    std::string e = values[1];

    if (!to_number(s, thefirst))
        return false;

    bool increment = (e[0] == '+');
    if (increment)
        e = e.substr(1);

    if (!to_number(e, thelast))
        return false;

    if (increment)
        thelast += thefirst - 1;

    first = thefirst;
    last = thelast;

    return true;
}
