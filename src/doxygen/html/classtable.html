    CYCLE now = _clk->now ();
    if (_cycle < now) {
        _wr_port.updateWireState (wire_name, axes_type == WRITE ? update_wire : false);
        _rd_port.updateWireState (wire_name, axes_type == READ ? update_wire : false);
