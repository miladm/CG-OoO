    {

        myArgv[i] = argv[i];
        if(strcmp("-t", argv[i])==0)
        {
            break;
        }
    }
    // next one is tool name
    i++;
    myArgv[i] = argv[i];
    *myArgc = i+1;
    // now skip all until --
    while (strcmp(argv[i],"--")!=0)
    {
        i++;
        ASSERTX(i<MAX_ARGV);
    }
    // now copy the -- and the rest
    while (i<argc)
    {
        ASSERTX(i<MAX_ARGV);
        myArgv[*myArgc] = argv[i];
        i++;
        (*myArgc)++;
    }
}
