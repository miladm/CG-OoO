bool wires::hasFreeWire ( ) 
                         {
    CYCLE now = _clk->now ();
    if (_cycle < now) {
        return true;
    } else if (_cycle == now) {
        bool result = (_num_free_wire > 0) ? true : false;
        if (!result) s_unavailable_cnt++;
        return result;
    }
    Assert (true == false && "should not have gotten here");
    return false;
}

        _cycle = now;
