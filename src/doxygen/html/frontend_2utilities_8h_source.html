00051 /* PrintHexWord:                                                      */
00052 /* prints a formatted version of a little endian memory word          */
00053 /* ------------------------------------------------------------------ */
00054 
00055 VOID PrintHexWord(ADDRINT addr, fstream& outfile) 
00056 {
00057     for (INT32 offset = sizeof(ADDRINT) - 1; offset >= 0; offset--) 
00058     {
00059         UINT32 val = (UINT32)*(UINT8*)(addr + offset);
00060         if (val < 0xf)        // formatting: pad single digits
00061         {
00062             outfile << "0" << flush;
00063         }
00064         outfile << hex << val << flush;
00065     }
00066 }

00124 /* RemoveToolArgs                                                     */
00125 /* Creates from the argv,argc pair new pair that does not contain the */
00126 /* the tool arguments                                                 */
00127 /* ------------------------------------------------------------------ */
00128 #define MAX_ARGV 256
00129 void RemoveToolArgs (int argc, char *argv[], int *myArgc, char *myArgv[])
00130 {
00131     int i;
00132     for (i=0; i<argc; i++)
00133     {
00134         // copy this argv into myArgv
00135         ASSERTX(i<MAX_ARGV);
00136         myArgv[i] = argv[i];
00137         if(strcmp("-t", argv[i])==0)
00138         {
00139             break;
00140         }
00141     }
00142     // next one is tool name
00143     i++;
00144     myArgv[i] = argv[i];
00145     *myArgc = i+1;
00146     // now skip all until --
00147     while (strcmp(argv[i],"--")!=0)
00148     {
00149         i++;
00150         ASSERTX(i<MAX_ARGV);
00151     }
00152     // now copy the -- and the rest
00153     while (i<argc)
00154     {
00155         ASSERTX(i<MAX_ARGV);
00156         myArgv[*myArgc] = argv[i];
00157         i++;
00158         (*myArgc)++;
00159     }
00160 }
